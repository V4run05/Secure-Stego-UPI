

### **Part 1: Understanding the Architecture**

**What you need to know:**

The project has 3 layers of authentication:

1. **Face Authentication** (Layer 1) — Backend verifies your face
2. **Dynamic PIN Challenge** (Layer 2) — Backend generates hidden PIN positions from encrypted data
3. **Transaction Authorization** (Layer 3) — Backend confirms the PIN and executes the transaction

**The Flow:**
```
Frontend UI → HTTP Request → Python Backend (api.py) → Response → Frontend Updates UI
```

**The Backend Exposes 4 Endpoints:**
- `POST /health` — Check if backend is alive
- `POST /auth/register` — Create account with face + PIN
- `POST /transaction/initiate` — Start payment (generates stego image)
- `POST /transaction/verify` — Confirm PIN and complete payment

### **Part 2: Create an API Service Layer (Best Practice)**

The key to clean integration is creating a **service file** that handles all backend communication. This keeps your components clean and your API logic in one place.

**File to create:** `frontend/src/app/services/api.ts`

```typescript
/**
 * API Service Layer
 * All backend communication happens here.
 * This is your "translator" between frontend and backend.
 */

// Configuration
const API_BASE_URL = import.meta.env.VITE_API_URL || "http://localhost:5000";

// Types for better code safety
interface RegisterPayload {
  user_id: string;
  face_image_b64: string;
  pin: string;
}

interface InitiateTransactionPayload {
  user_id: string;
  face_image_b64: string;
  amount_rupees: number;
  recipient_upi: string;
}

interface VerifyTransactionPayload {
  tx_id: string;
  pin_digits: Record<string, string>; // e.g., {"1": "4", "3": "7", "5": "2"}
}

interface ApiResponse<T> {
  success: boolean;
  data?: T;
  error?: string;
  details?: any;
}

// Timeout helper - prevents hanging requests
function createAbortSignal(timeoutMs = 30000) {
  const controller = new AbortController();
  setTimeout(() => controller.abort(), timeoutMs);
  return controller.signal;
}

// Main API client
export const apiClient = {
  /**
   * Check if backend is running
   * Use this as a "heartbeat" to verify connectivity
   */
  async checkHealth() {
    try {
      const response = await fetch(`${API_BASE_URL}/health`, {
        method: "GET",
        signal: createAbortSignal(5000), // Short timeout for health check
      });

      if (!response.ok) {
        throw new Error(`Health check failed: ${response.status}`);
      }

      return {
        success: true,
        data: await response.json(),
      };
    } catch (error) {
      return {
        success: false,
        error: error instanceof Error ? error.message : "Health check failed",
      };
    }
  },

  /**
   * Register a new user with face and PIN
   * This is called during account setup
   */
  async register(payload: RegisterPayload) {
    try {
      const response = await fetch(`${API_BASE_URL}/auth/register`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload),
        signal: createAbortSignal(),
      });

      const data = await response.json();

      if (!response.ok) {
        return {
          success: false,
          error: data.reason || "Registration failed",
          details: data,
        };
      }

      return { success: true, data };
    } catch (error) {
      return {
        success: false,
        error: error instanceof Error ? error.message : "Network error during registration",
      };
    }
  },

  /**
   * Initiate a transaction (Step 1 of payment)
   * Returns: stego_image_b64, pin positions, and transaction ID
   */
  async initiateTransaction(payload: InitiateTransactionPayload) {
    try {
      const response = await fetch(`${API_BASE_URL}/transaction/initiate`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload),
        signal: createAbortSignal(),
      });

      const data = await response.json();

      if (!response.ok) {
        return {
          success: false,
          error: data.error || "Failed to initiate transaction",
          attempts_remaining: data.attempts_remaining,
          details: data,
        };
      }

      return { success: true, data };
    } catch (error) {
      return {
        success: false,
        error: error instanceof Error ? error.message : "Network error",
      };
    }
  },

  /**
   * Verify PIN (Step 2 of payment)
   * This completes the transaction if PIN is correct
   */
  async verifyTransaction(payload: VerifyTransactionPayload) {
    try {
      const response = await fetch(`${API_BASE_URL}/transaction/verify`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload),
        signal: createAbortSignal(),
      });

      const data = await response.json();

      if (!response.ok) {
        return {
          success: false,
          error: data.reason || "PIN verification failed",
          attempts_remaining: data.attempts_remaining,
          details: data,
        };
      }

      return { success: true, data };
    } catch (error) {
      return {
        success: false,
        error: error instanceof Error ? error.message : "Network error",
      };
    }
  },
};
```

### **Part 3: Create Hooks for Common Tasks**

Hooks let you reuse logic across components. Create: `frontend/src/app/hooks/useBackendCall.ts`

```typescript
/**
 * Custom React Hook for Backend Calls
 * Handles loading, errors, and retries automatically
 */

import { useState, useCallback } from "react";

interface UseBackendCallOptions {
  maxRetries?: number;
  onError?: (error: string) => void;
  onSuccess?: (data: any) => void;
}

export function useBackendCall(
  apiFunction: (...args: any[]) => Promise<any>,
  options: UseBackendCallOptions = {}
) {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [data, setData] = useState<any>(null);
  const [retryCount, setRetryCount] = useState(0);

  const { maxRetries = 2, onError, onSuccess } = options;

  const execute = useCallback(
    async (...args: any[]) => {
      setLoading(true);
      setError(null);

      try {
        const result = await apiFunction(...args);

        if (result.success) {
          setData(result.data);
          onSuccess?.(result.data);
        } else {
          throw new Error(result.error || "Unknown error");
        }
      } catch (err) {
        const errorMessage = err instanceof Error ? err.message : "Unknown error";

        // Automatic retry logic
        if (retryCount < maxRetries) {
          setRetryCount(retryCount + 1);
          // Wait before retrying (exponential backoff)
          await new Promise((resolve) =>
            setTimeout(resolve, Math.pow(2, retryCount) * 1000)
          );
          return execute(...args);
        }

        setError(errorMessage);
        onError?.(errorMessage);
      } finally {
        setLoading(false);
      }
    },
    [apiFunction, maxRetries, retryCount, onError, onSuccess]
  );

  const reset = useCallback(() => {
    setError(null);
    setData(null);
    setRetryCount(0);
  }, []);

  return { loading, error, data, execute, reset };
}
```

### **Part 6: Error Handling & Fallback Strategy**

**Common scenarios and how to handle them:**

| Scenario | What Happens | Fallback |
|----------|--------------|----------|
| Backend down | `apiClient` returns `success: false` | Show demo mode, use offline UI |
| Network timeout | `fetch` aborts after 30s | Retry up to 2 times, then error |
| Invalid face image | Backend rejects in `face_auth.py` | Show camera retry message |
| Incorrect PIN | Backend returns `attempts_remaining` | Show remaining attempts |
| Network flaky | Request sometimes succeeds | Retry with exponential backoff |

**Generic error boundary component:**

Create: `frontend/src/app/components/ErrorBoundary.tsx`

```typescript
import { ReactNode, Component } from "react";
import { AlertCircle } from "lucide-react";

interface Props {
  children: ReactNode;
}

interface State {
  hasError: boolean;
  error: string;
}

export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false, error: "" };
  }

  static getDerivedStateFromError(error: Error) {
    return { hasError: true, error: error.message };
  }

  componentDidCatch(error: Error) {
    console.error("Error caught:", error);
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="min-h-screen bg-red-950 text-white flex items-center justify-center p-4">
          <div className="text-center">
            <AlertCircle className="w-16 h-16 mx-auto mb-4 text-red-400" />
            <h1 className="text-2xl mb-2">Something went wrong</h1>
            <p className="text-red-200 mb-4">{this.state.error}</p>
            <button
              onClick={() => window.location.reload()}
              className="bg-red-600 hover:bg-red-700 px-4 py-2 rounded"
            >
              Reload Page
            </button>
          </div>
        </div>
      );
    }

    return this.props.children;
  }
}
```

### **Part 7: Backend Server Setup (Python)**

Create: `run_backend.py` in the root directory

```python
"""
Simple Flask wrapper to expose the API
"""
import os
from flask import Flask, request, jsonify
from flask_cors import CORS

from api import SecureStegoUPI

# Initialize Flask app
app = Flask(__name__)
CORS(app)  # Enable frontend to call from different origin

# Load the API (use untrained mode for development)
try:
    api = SecureStegoUPI.from_checkpoint("checkpoints/checkpoint_final.pt")
except:
    print("⚠️ Checkpoint not found - using untrained mode")
    api = SecureStegoUPI.untrained()

# ── Routes ──────────────────────────────────────────────────────────────

@app.route("/health", methods=["GET"])
def health():
    """Check if backend is alive"""
    return jsonify(api.health())

@app.route("/auth/register", methods=["POST"])
def register():
    """Register user with face + PIN"""
    data = request.json
    result = api.register_user(
        user_id=data.get("user_id"),
        face_image_b64=data.get("face_image_b64"),
        pin=data.get("pin")
    )
    return jsonify(result), 200 if result.get("success") else 400

@app.route("/transaction/initiate", methods=["POST"])
def initiate():
    """Start a transaction"""
    data = request.json
    result = api.initiate_transaction(
        user_id=data.get("user_id"),
        face_image_b64=data.get("face_image_b64"),
        amount_rupees=data.get("amount_rupees"),
        recipient_upi=data.get("recipient_upi")
    )
    return jsonify(result), 200 if result.get("tx_id") else 400

@app.route("/transaction/verify", methods=["POST"])
def verify():
    """Verify PIN and complete transaction"""
    data = request.json
    result = api.verify_transaction(
        tx_id=data.get("tx_id"),
        pin_digits=data.get("pin_digits")
    )
    return jsonify(result), 200 if result.get("authorized") else 400

if __name__ == "__main__":
    port = int(os.environ.get("PORT", 5000))
    app.run(debug=True, host="0.0.0.0", port=port)
```
